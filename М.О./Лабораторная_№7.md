def load_reviews_from_dir(path):
    reviews = []  # –ø—É—Å—Ç–æ–π —Å–ø–∏—Å–æ–∫ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –æ—Ç–∑—ã–≤–æ–≤
    for filename in os.listdir(path):  # –ø–µ—Ä–µ–±–æ—Ä –≤—Å–µ—Ö —Ñ–∞–π–ª–æ–≤
        if filename.endswith(".txt"):  # –ø—Ä–æ–≤–µ—Ä–∫–∞, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —Ñ–∞–π–ª —Ç–µ–∫—Å—Ç–æ–≤—ã–º
            with open(os.path.join(path, filename), encoding="utf-8") as f:  # –æ—Ç–∫—Ä—ã—Ç–∏–µ —Ñ–∞–π–ª–∞ —Å –∫–æ–¥–∏—Ä–æ–≤–∫–æ–π UTF-8
                reviews.append(f.read())  # —á—Ç–µ–Ω–∏–µ —Å–æ–¥–µ—Ä–∂–∏–º–æ–≥–æ —Ñ–∞–π–ª–∞ –∏ –¥–æ–±–∞–≤–ª–µ–Ω–∏–µ –µ–≥–æ –≤ —Å–ø–∏—Å–æ–∫ –æ—Ç–∑—ã–≤–æ–≤

```python
def preprocess(text):
    text = text.lower()  # –ø—Ä–∏–≤–µ–¥–µ–Ω–∏–µ —Ç–µ–∫—Å—Ç–∞ –∫ –Ω–∏–∂–Ω–µ–º—É —Ä–µ–≥–∏—Å—Ç—Ä—É
    text = re.sub(f"[{string.punctuation}]", "", text)  # —É–¥–∞–ª–µ–Ω–∏–µ –∑–Ω–∞–∫–æ–≤ –ø—Ä–µ–ø–∏–Ω–∞–Ω–∏—è
    text = re.sub("\\d+", "", text)  # —É–¥–∞–ª–µ–Ω–∏–µ —Ü–∏—Ñ—Ä
    tokens = word_tokenize(text)  # —Ç–æ–∫–µ–Ω–∏–∑–∞—Ü–∏—è —Ç–µ–∫—Å—Ç–∞ –Ω–∞ –æ—Ç–¥–µ–ª—å–Ω—ã–µ —Å–ª–æ–≤–∞
    processed_tokens = []  # –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è —Å–ø–∏—Å–∫–∞ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –æ–±—Ä–∞–±–æ—Ç–∞–Ω–Ω—ã—Ö —Ç–æ–∫–µ–Ω–æ–≤
    for word in tokens:
        if word not in stop_words:  # –ø—Ä–æ–≤–µ—Ä—è—Ç –Ω–µ —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —Å–ª–æ–≤–æ —Å—Ç–æ–ø-—Å–ª–æ–≤–æ–º
            parsed_word = morph.parse(word)[0]  # –º–æ—Ä—Ñ–æ–ª–æ–≥–∏—á–µ—Å–∫–∏–π —Ä–∞–∑–±–æ—Ä
            lemma = parsed_word.normal_form  # –ø–æ–ª—É—á–µ–Ω–∏–µ –ª–µ–º–º—ã —Å–ª–æ–≤–∞
            pos = parsed_word.tag.POS  # –ø–æ–ª—É—á–µ–Ω–∏–µ —á–∞—Å—Ç–∏ —Ä–µ—á–∏
            processed_tokens.append(f"{lemma}_{pos}")  # –¥–æ–±–∞–≤–ª–µ–Ω–∏–µ –ª–µ–º–º—ã –∏ —á–∞—Å—Ç–∏ —Ä–µ—á–∏ –≤ —Å–ø–∏—Å–æ–∫
    return processed_tokens
```

def get_top_tokens(reviews, n=100):  # —Ñ—É–Ω–∫—Ü–∏—è, –∫–æ—Ç–æ—Ä–∞—è –ø—Ä–∏–Ω–∏–º–∞–µ—Ç —Å–ø–∏—Å–æ–∫ –æ—Ç–∑—ã–≤–æ–≤ –∏ 100 –¥–ª—è –≤—ã–±–æ—Ä–∞ —Å–∞–º—ã—Ö –ø–æ–ø—É–ª—è—Ä–Ω—ã—Ö —Å–ª–æ–≤
    tokens = [token for review in reviews for token in review]  # —Å–æ–∑–¥–∞–Ω–∏–µ —Å–ø–∏—Å–∫–∞ —Ç–æ–∫–µ–Ω–æ–≤, –ø—Ä–æ—Ö–æ–¥—è –ø–æ –∫–∞–∂–¥–æ–º—É –æ—Ç–∑—ã–≤—É –∏ –∏–∑–≤–ª–µ–∫–∞—è –∏–∑ –Ω–µ–≥–æ –≤—Å–µ —Å–ª–æ–≤–∞
    return [word for word, _ in Counter(tokens).most_common(n)]  # —Å–ø–∏—Å–æ–∫ –∏–∑ n —Å–∞–º—ã—Ö —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–Ω—ã—Ö —Å–ª–æ–≤

```python
def plot_tsne(model, tokens, title):  # —Ñ—É–Ω–∫—Ü–∏—è –¥–ª—è —Å–ø–∏—Å–∫–∞ —Ç–æ–∫–µ–Ω–æ–≤ –∏ –∑–∞–≥–æ–ª–æ–≤–æ–∫ –¥–ª—è –≥—Ä–∞—Ñ–∏–∫–∞
    vectors = [model.wv[token] for token in tokens if token in model.wv]  # –∏–∑–≤–ª–µ—á–µ–Ω–∏–µ –≤–µ–∫—Ç–æ—Ä–æ–≤ —Å–ª–æ–≤ –∏–∑ –º–æ–¥–µ–ª–∏ –¥–ª—è —Ç–æ–∫–µ–Ω–æ–≤=
    labels = [token for token in tokens if token in model.wv]  # —Å–ø–∏—Å–æ–∫ –º–µ—Ç–æ–∫

    vectors = np.array(vectors)  # —Å–ø–∏—Å–æ–∫ –≤–µ–∫—Ç–æ—Ä–æ–≤

    tsne = TSNE(n_components=2, random_state=42)  # t-SNE —Å 2 –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–∞–º–∏ –∏ —Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω—ã–º —Å–æ—Å—Ç–æ—è–Ω–∏–µ–º –≥–µ–Ω–µ—Ä–∞—Ç–æ—Ä–∞ —Å–ª—É—á–∞–π–Ω—ã—Ö —á–∏—Å–µ–ª
    reduced_vectors = tsne.fit_transform(vectors)  # t-SNE –¥–ª—è —Å–Ω–∏–∂–µ–Ω–∏—è —Ä–∞–∑–º–µ—Ä–Ω–æ—Å—Ç–∏ –≤–µ–∫—Ç–æ—Ä–æ–≤

    # –ü–æ—Å—Ç—Ä–æ–µ–Ω–∏–µ –≥—Ä–∞—Ñ–∏–∫–∞
    plt.figure(figsize=(12, 8))
    sns.scatterplot(x=reduced_vectors[:, 0], y=reduced_vectors[:, 1])

    for i, label in enumerate(labels):
        plt.text(reduced_vectors[i, 0], reduced_vectors[i, 1], label, fontsize=9)

    plt.title(title).
    plt.show()

```
